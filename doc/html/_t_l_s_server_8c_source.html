11 handshake * make_server_hello(TLS_parameters *TLS_param, handshake_hello *client_hello){
12 
13  // Initialize server hello (without SessionID)
14  session_id *session= malloc(sizeof(session_id));

23  int choosen_suite_num = rand()%(client_hello->cipher_suite_len/2); // Specify the number of supported cipher suite
24  cipher_suite_t choosen_suite = get_cipher_suite( client_hello->cipher_suites[choosen_suite_num].cipher_id );

26  server_hello->cipher_suites = malloc(sizeof(cipher_suite_t));
27  *(server_hello->cipher_suites) = choosen_suite;
28 
29  // Insert server hello into handshake packet
30  handshake *server_hello_h = malloc(sizeof(handshake));
31  server_hello_h->type = SERVER_HELLO;
32  server_hello_h->message = NULL;
33  server_hello_h->length = 0;
34  serialize_client_server_hello(server_hello, &(server_hello_h->message), &(server_hello_h->length), SERVER_MODE);
35 
36  // Save parameters
37  TLS_param->cipher_suite = choosen_suite;
38  memcpy(TLS_param->server_random,&(server_hello->random.UNIX_time), 4);
39  memcpy(TLS_param->server_random+4, server_hello->random.random_bytes, 28);
40 
41  // Clean up
42  free_hello(server_hello);
43  free(session);
44

113 DHE_server_key_exchange * make_DHE_server_key_exchange(TLS_parameters *TLS_param){
114 
115  // Diffie-Hellman server key exchange
116  // Generate ephemeral Diffie-Hellman parameters
117  DH *privkey;
118  int codes;
119  if((privkey = DH_new()) == NULL)
120  printf("\nError in DH_new\n");
121  if(DH_generate_parameters_ex(privkey, 1024, DH_GENERATOR_2 , NULL) != 1)
122  printf("\nError in DH_generate_parameters\n");
123  if(DH_check(privkey, &codes) != 1)
124  printf("\nError in DH_check\n");
125  if(codes != 0)
126  printf("\nDH_check failed\n");
127  // Generate the public and private keys pair
128  if(DH_generate_key(privkey) != 1)
129  printf("Error in DH_generate_key\n");
130 
131  // Set server key exchange parameters
132  DHE_server_key_exchange *server_key_ex = malloc(sizeof(DHE_server_key_exchange));
133  server_key_ex->g = BN_dup(privkey->g);
134  server_key_ex->p = BN_dup(privkey->p);
135  server_key_ex->pubKey = BN_dup(privkey->pub_key);
136 
137  // Set hash algorithm and authentication
138  server_key_ex->sign_hash_alg = TLS_param->cipher_suite.hash+(TLS_param->cipher_suite.au<<8); // 0x0106 // Already rotated
139 
140  // Add signature
141  sign_DHE_server_key_ex(TLS_param->client_random, TLS_param->server_random, server_key_ex, TLS_param->cipher_suite.au);
142 
143  // Save parameters
144  TLS_param->server_key_ex = server_key_ex;
145  TLS_param->private_key = BN_dup(privkey->priv_key);
146 
147  // Clean up
148  DH_free(privkey);
149 
150  return server_key_ex;
151 }

154 
155  // Elliptic cruve Diffie-Hellman server key exchange
156  // Generate ephemeral Diffie-Hellman parameters
157  EC_KEY *key;
158  uint16_t curve_name = NID_secp256k1;
159  // Create an Elliptic Curve Key object and set it up to use the ANSI X9.62 Prime 256v1 curve
160  if((key = EC_KEY_new_by_curve_name(curve_name)) == NULL)
161  printf("\nError setting EC parameters\n");
162  // Generate the private and public keys
163  if(EC_KEY_generate_key(key) != 1)
164  printf("\nError in generate EC keys\n");
165 
166  // Set server key exchange parameters
167  ECDHE_server_key_exchange *server_key_ex = malloc(sizeof(ECDHE_server_key_exchange));
168  server_key_ex->named_curve = curve_name;
169  server_key_ex->pub_key = BN_new();
170  EC_POINT_point2bn(EC_KEY_get0_group(key), EC_KEY_get0_public_key(key), POINT_CONVERSION_UNCOMPRESSED, server_key_ex->pub_key, NULL);
171 
172  // Set hash algorithm and authentication
173  server_key_ex->sign_hash_alg = TLS_param->cipher_suite.hash+(TLS_param->cipher_suite.au<<8); // Already rotated
174 
175  // Add signature

177 
178  // Save parameters
179  TLS_param->server_key_ex = server_key_ex;
180  TLS_param->private_key = BN_dup(EC_KEY_get0_private_key(key));

182  // Clean up
