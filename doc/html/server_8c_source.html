213 void compute_set_master_key_RSA(client_key_exchange *client_key_exchange) {
214  //get private key from file
215  RSA *privateKey = NULL;
216  FILE *fp;
217 
218  if(NULL != (fp= fopen("../certificates/serverRSA.key", "r")) )
219  {
220  privateKey=PEM_read_RSAPrivateKey(fp,NULL,NULL,NULL);
221  if(privateKey==NULL)
222  {
223  printf("\nerror in retrieve private key");
224  exit(-1);
225  }
226  }
227  fclose(fp);
228  unsigned char *pre_master_key=malloc(100);
229  if(!RSA_private_decrypt(client_key_exchange->key_length, client_key_exchange->key, pre_master_key, privateKey, RSA_PKCS1_PADDING))
230  {
231  printf("Error decrypt\n");
232  exit(-1);
233  }
234 
235  //make master key
236  unsigned char seed[64];
237  memcpy(seed, TLS_param.client_random, 32);
238  memcpy(seed+32, TLS_param.server_random, 32);
239 
240  const EVP_MD *hash_function = get_hash_function(TLS_param.cipher_suite.hash);
241  TLS_param.master_secret_len = 48;
242  PRF(hash_function, pre_master_key, 48, "master secret", seed, 64, TLS_param.master_secret_len, &TLS_param.master_secret);
243  RSA_free(privateKey);
244

246  free(client_key_exchange->key);

279 void compute_set_master_key_ECDHE(client_key_exchange *cliet_public){
280  ECDHE_server_key_exchange *server_key_exchange = (ECDHE_server_key_exchange *) TLS_param.server_key_ex;
281  EC_KEY *key = EC_KEY_new_by_curve_name(server_key_exchange->named_curve);
282 
283  //get and set public key
284  BIGNUM *pub_key = BN_bin2bn(cliet_public->key, cliet_public->key_length, NULL);
285  EC_POINT *pub_key_point = EC_POINT_bn2point(EC_KEY_get0_group(key), pub_key, NULL, NULL);
286  EC_KEY_set_public_key(key, pub_key_point);
287  EC_POINT_free(pub_key_point);
288  BN_free(pub_key);
289 
290  //set private key
291  EC_KEY_set_private_key(key, TLS_param.private_key);
292 
293  // compute master secret
294  int field_size, pre_master_len;
295  unsigned char *pre_master;
296 
297  /* Calculate the size of the buffer for the shared secret */
298  field_size = EC_GROUP_get_degree(EC_KEY_get0_group(key));
299  pre_master_len = (field_size+7)/8;
300 
301  /* Allocate the memory for the shared secret */
302  pre_master = malloc(sizeof(unsigned char)*pre_master_len);
303 
304  /* Derive the shared secret */
305  TLS_param.master_secret_len = ECDH_compute_key(pre_master, pre_master_len, EC_KEY_get0_public_key(key), key, NULL);
306 
307  // Derive master key
308  unsigned char seed[64];
309  memcpy(seed, TLS_param.client_random, 32);
310  memcpy(seed+32, TLS_param.server_random, 32);
311  const EVP_MD *hash_function = get_hash_function(TLS_param.cipher_suite.hash);
312  TLS_param.master_secret_len = 48;
313 
314  //compute and set pre master key
