108  TLS_param->master_secret_len = 48;

276  EVP_MD_CTX_destroy(mdctx);

278  unsigned char *finished_message = NULL;
279  int finished_message_len = 12;
280  PRF(hash_function, TLS_param->master_secret, TLS_param->master_secret_len, "client finished", md_value, md_len, finished_message_len, &finished_message);
