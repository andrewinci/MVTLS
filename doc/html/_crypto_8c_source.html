15 void PRF(const EVP_MD *hash, unsigned char *secret, int secret_len, char *label, unsigned char *seed, int seed_len, int result_len, unsigned char **result){
16  int buffer_size = ((1+result_len/hash->md_size)*hash->md_size);
17  unsigned char *buff = malloc(buffer_size);
18  int label_len = (int)strlen(label);
19  *result = buff;
20 
21  //compute p_hash(secret,seed)
22  //secret is equal to secret
23  //seed is equal to label concatenate with seed
24  unsigned char *seed_p = malloc(label_len+seed_len);
25  memcpy(seed_p, label, label_len);
26  memcpy(seed_p+label_len, seed, seed_len);
27 
28  //compute A_i
29  int tot_len = 0;
30  unsigned int a_len = label_len+seed_len;
31  unsigned char *a = seed_p;
32  while (tot_len<result_len) {
33  unsigned char *temp = NULL;
34  temp = HMAC(hash, secret, secret_len, a, a_len, NULL, &a_len);
35  a = temp;
36  memcpy(buff+tot_len, a, a_len);
37  tot_len+=a_len;
38  }
39  free(seed_p);
40 }
